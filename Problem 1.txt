    char ch=input[0];
    if((ch>='A'&&ch<='Z')||(ch>='a'&&ch<='z')||ch=='_')
        {

        int i=1;
        while (input[i]!='\0')
            {
            ch=input[i];
            if (!((ch >= 'A' && ch <= 'Z') ||
                  (ch >= 'a' && ch <= 'z') ||
                  (ch >= '0' && ch <= '9') ||
                  ch == '_'))
                  {
                cout<<"Not Identifier" << endl;
                return 0;
            }
            i++;
        }
        cout<<"Identifier"<<endl;
    }
    else
    {
